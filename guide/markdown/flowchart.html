# 流程图支持 V0.0.10+

让你的 VuePress 站点中的 Markdown 文件支持流程图。

# 配置

module.exports = {
  themeConfig: {
    markdown: {
      // 启用流程图功能
      flowchart: true
    }
  }
};

@flowstart [preset]

<!-- 放置你的流程图代码 -->

@flowend
